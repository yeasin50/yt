Pyramid Patterns in C

Pyramid Top to Bottom Print pyramid patterns from top to bottom. #include <stdio.h> int main() { int n; printf("Enter Number of rows:"); scanf("%d", &n); for (int i = 0; i < n; i++) { for (int s = 0; s < n - i - 1; s++) printf(" "); for (int j = 0; j < 2 * i + 1; j++) { printf("*"); } printf("\n"); } return 0; } Upside Pyramid Print upside-down pyramid patterns step by step. #include <stdio.h> int main() { int n; printf("Enter Number of rows:"); scanf("%d", &n); for (int i = 0; i < n; i++) { for (int s = 0; s < i; s++) printf(" "); for (int j = 0; j < 2 * (n - i) - 1; j++) { printf("*"); } printf("\n"); } return 0; } Hollow Upside Pyramid #include <stdio.h> int main() { int n ; printf("Enter Number of rows:"); scanf("%d", &n); for (int i=0; i< n; i++){ for(int s=0; s<i;s++) printf(" "); for(int j=0; j<2*(n-i)-1; j++){ if(j==0||i==0 ||j==2*(n-i)-2 ) printf("*"); else printf(" "); } printf("\n"); } return 0; } Pyramid with Space Top to bottom #include <stdio.h> int main() { int n ; printf("Enter Number of rows:"); scanf("%d", &n); for (int i=0; i< n; i++){ for(int s=0; s<n-i-1;s++) printf(" "); for(int j=0; j<2*i+1; j++){ if(j==0||i==n-1 ||j==2*i) printf("*"); else printf(" "); } printf("\n"); } return 0; }
